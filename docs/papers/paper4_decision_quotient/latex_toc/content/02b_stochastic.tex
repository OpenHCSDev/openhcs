\section{Stochastic Decision Problems}\label{sec:stochastic}

We extend the static decision problem to include a probability distribution over states.

\subsection{Stochastic Decision Problem}

\begin{definition}[Stochastic Decision Problem]\label{def:stochastic-decision-problem}
A \emph{stochastic decision problem with coordinate structure} is a tuple $\mathcal{D}_S = (A, X_1, \ldots, X_n, U, P)$ where:
\begin{itemize}
\item $A$ is a finite set of \emph{actions}
\item $X_1, \ldots, X_n$ are finite \emph{coordinate spaces}
\item $S = X_1 \times \cdots \times X_n$ is the \emph{state space}
\item $U : A \times S \to \mathbb{Q}$ is the \emph{utility function}
\item $P \in \Delta(S)$ is a \emph{distribution over states}
\end{itemize}
\end{definition}

\begin{definition}[Distribution-Conditional Sufficiency]\label{def:stochastic-sufficient}
A coordinate set $I \subseteq \{1, \ldots, n\}$ is \emph{stochastically sufficient} for $\mathcal{D}_S$ if:
\[
P(s \mid s_I) \otimes U \text{ determines the same optimal action as } P(s) \otimes U.
\]
Equivalently, for all $s, s' \in S$ with $s_I = s'_I$:
\[
\arg\max_{a \in A} \mathbb{E}_{t \sim P(\cdot \mid s_I)}[U(a, t)] = \arg\max_{a \in A} \mathbb{E}_{t \sim P}[U(a, t)].
\]
\end{definition}

\begin{definition}[Expected-Value Sufficiency (Alternative)]\label{def:expected-value-sufficient}
A coordinate set $I$ is \emph{expected-value sufficient} if:
\[
\forall a, a' : \mathbb{E}_{s \sim P}[U(a,s) \mid s_I] \geq \mathbb{E}_{s \sim P}[U(a',s) \mid s_I] \implies a \in \Opt_P.
\]
where $\Opt_P = \arg\max_{a \in A} \mathbb{E}_{s \sim P}[U(a,s)]$.
\end{definition}

\begin{decision}\label{dec:stochastic-sufficiency-check}
\textbf{STOCHASTIC-SUFFICIENCY-CHECK}: Given stochastic decision problem $\mathcal{D}_S = (A, S, U, P)$ and coordinate set $I$, is $I$ stochastically sufficient?
\end{decision}

\begin{decision}\label{dec:stochastic-minsuff}
\textbf{STOCHASTIC-MINIMUM-SUFFICIENT-SET}: Given $\mathcal{D}_S$, find the minimal $I \subseteq \{1, \ldots, n\}$ that is stochastically sufficient.
\end{decision}

\begin{decision}\label{dec:stochastic-anchor-sufficiency-check}
\textbf{STOCHASTIC-ANCHOR-SUFFICIENCY-CHECK}: Given stochastic decision problem $\mathcal{D}_S$ and coordinate set $I$, does there exist an anchor fiber whose conditional-optimal action is singleton and fiber-stable under agreement on $I$?
\end{decision}

\begin{enumerate}
\item \textbf{Membership in \PP}: Given $\mathcal{D}_S$ and $I$, we must check whether knowing $s_I$ determines the optimal action. This reduces to comparing expected utilities conditioned on $s_I$ versus unconditioned. For each action pair $(a, a')$, we compare:
\[
\mathbb{E}_{s \sim P}[U(a,s) \mid s_I] \geq \mathbb{E}_{s \sim P}[U(a',s) \mid s_I].
\]
This requires counting weighted by $P$, which is in \PP.

\item \textbf{\PP-hardness via MAJSAT reduction}: Given Boolean formula $\varphi$ on $n$ variables, construct:
\begin{itemize}
\item $S = \{0,1\}^n$ (state space = assignments)
\item $P$ = uniform distribution over $S$
\item $A = \{\text{accept}, \text{reject}\}$
\item $U(\text{accept}, s) = \varphi(s)$, $U(\text{reject}, s) = 1/2$
\item $I = \emptyset$
\end{itemize}
Then $\emptyset$ is stochastically sufficient iff the optimal action is determined without seeing any coordinates:
\[
\emptyset \text{ sufficient } \iff \mathbb{E}_{s \sim P}[\varphi(s)] \geq 1/2 \text{ or } < 1/2.
\]
This is exactly MAJSAT, which is \PP-complete.
\end{enumerate}

\leanmeta{\LH{CC58},
\LH{CC55}}

\begin{proof}
For membership: We reduce to a \PP predicate. For each $a \in A$, compute $\mathbb{E}_P[U(a,s)]$ and compare. The comparison involves summing over $s \in S$ weighted by $P(s)$, which is polynomial in $|S|$ and the representation of $P$. The decision is whether the argmax is unique or whether knowing $s_I$ changes it; this is a majority-vote comparison, hence in \PP.

For hardness: The reduction from MAJSAT constructs the uniform distribution. If $\varphi$ is true on $\geq 1/2$ of assignments, the expected utility of accepting is $\geq 1/2$, making accept optimal regardless of state observation. If $< 1/2$, reject is optimal. Thus $\emptyset$ is sufficient exactly when MAJSAT is true.
\end{proof}

\begin{example}[Stochastic Weather]
Consider stochastic umbrella decision:
\begin{itemize}
\item $A = \{\text{carry}, \text{don't carry}\}$
\item $S = \{\text{rain}, \text{no rain}\}$
\item $P(\text{rain}) = 0.3$, $P(\text{no rain}) = 0.7$
\item $U(\text{carry}, \text{rain}) = 2$, $U(\text{carry}, \text{no rain}) = -1$
\item $U(\text{don't carry}, \text{rain}) = -2$, $U(\text{don't carry}, \text{no rain}) = 0$
\end{itemize}
Expected utility: $E_P[U(\text{carry})] = 0.3(2) + 0.7(-1) = -0.1$, $E_P[U(\text{don't carry})] = 0.3(-2) + 0.7(0) = -0.6$. So carry is optimal. If we condition on forecast information, we check whether the conditional distribution changes the argmax.
\end{example}

\subsection{Potential and Physical Bounds}

\begin{definition}[Potential Grounding]
Let $\Phi : A \times S \to \mathbb{R}$ with grounding equation
\[
U(a,s) = -\Phi(a,s).
\]
Define expected action-potential
\[
\mathbb{E}[\Phi\mid a] := \sum_{s\in S} P(s)\,\Phi(a,s).
\]
\end{definition}

\leanmeta{\LH{DC30},\LH{DC32}.}

\begin{claim}{Expected-Utility/Potential Duality}{prop:stochastic-potential-duality}
Under the grounding equation, for every action $a$:
\[
\mathbb{E}_{s\sim P}[U(a,s)] = -\,\mathbb{E}[\Phi\mid a],
\]
and for all $a,a'$:
\[
\mathbb{E}[U\mid a]\le \mathbb{E}[U\mid a']
\iff
\mathbb{E}[\Phi\mid a']\le \mathbb{E}[\Phi\mid a].
\]
\claimstamp{P\ref{prop:stochastic-potential-duality}}{PP}
\leanmeta{\LH{DC31},\LH{DC33},\LH{DC34}.}
\end{claim}

\leanmeta{\LH{DC31},\LH{DC33},\LH{DC34}.}

\begin{claim}{Landauer Floor and State-Cardinality Specialization}{prop:stochastic-landauer-floor}
For $T\ge 0$, the Landauer floor is nonnegative and monotone in erased bits:
\[
E_{\mathrm{L}}(b,T)\ge 0,\qquad
b_1\le b_2 \implies E_{\mathrm{L}}(b_1,T)\le E_{\mathrm{L}}(b_2,T).
\]
At the room-temperature specialization used here:
\[
\mathrm{thermodynamicCost}(\mathcal{D}_S)=\mathrm{landauerConstant}\cdot |S|.
\]
\claimstamp{P\ref{prop:stochastic-landauer-floor}}{AR,H=cost-growth}
\leanmeta{\LH{DC35},\LH{DC36},\LH{DC37}.}
\end{claim}

\subsection{Tractable Subcases}

\begin{enumerate}
\item \textbf{Product distributions}: $P = P_1 \otimes \cdots \otimes P_n$. Then
\[
\mathbb{E}_{s \sim P}[U(a,s) \mid s_I] = \prod_{i \in I} P_i(s_i) \cdot \mathbb{E}_{s \sim P}[U(a,s) \mid s_I],
\]
so expected utility decomposes. This reduces stochastic sufficiency to static sufficiency per coordinate.

\item \textbf{Bounded support}: $|\text{supp}(P)| \leq k$. Enumeration is $O(k \cdot |A|)$, polynomial.

\item \textbf{Log-concave distributions}: For ordered state spaces, conditional expectations are monotone and computable via convex optimization.
\end{enumerate}

\begin{claim}{Tractability under product distributions}{prop:stochastic-product-tractable}
If $P = P_1 \otimes \cdots \otimes P_n$ is a product distribution, then STOCHASTIC-SUFFICIENCY-CHECK reduces to checking static sufficiency on each marginal.
\claimstamp{P\ref{prop:stochastic-product-tractable}}{PD}
\leanmeta{\LH{CC60}.}
\end{claim}

\leanmeta{\LH{CC60}}

\begin{claim}{Tractability under bounded support}{prop:stochastic-bounded-support}
If $|\text{supp}(P)| \leq k$, then STOCHASTIC-SUFFICIENCY-CHECK is solvable in $O(k \cdot |A| \cdot n)$ time.
\claimstamp{P\ref{prop:stochastic-bounded-support}}{BS}
\leanmeta{\LH{CC59}.}
\end{claim}

\subsection{Bridge from Static}

Paper 4 proved that static sufficiency does NOT imply stochastic sufficiency (\LH{CC54}). A counterexample exists where $I$ is sufficient for $(A,S,U)$ but not for $(A,S,U,P)$.

However, under product distributions, static sufficiency transfers:

\begin{claim}{Static to stochastic transfer}{prop:static-stochastic-transfer}
If $P = P_1 \otimes \cdots \otimes P_n$ is a product distribution and $I$ is statically sufficient for $(A,S,U)$, then $I$ is stochastically sufficient for $(A,S,U,P)$.
\claimstamp{P\ref{prop:static-stochastic-transfer}}{PD}
\leanmeta{\LH{CC52}.}
\end{claim}
