\section{Sequential Decision Problems}\label{sec:sequential}

We further extend to sequential decision-making with transitions and observations.

\subsection{Sequential Decision Problem (POMDP)}

\begin{definition}[Sequential Decision Problem]\label{def:sequential-decision-problem}
A \emph{sequential decision problem with coordinate structure} is a tuple $\mathcal{D}_{SEQ} = (A, X_1, \ldots, X_n, U, T, O)$ where:
\begin{itemize}
\item $A$ is a finite set of \emph{actions}
\item $X_1, \ldots, X_n$ are finite \emph{coordinate spaces}
\item $S = X_1 \times \cdots \times X_n$ is the \emph{state space}
\item $U : A \times S \to \mathbb{Q}$ is the \emph{stage utility}
\item $T : A \times S \to \Delta(S)$ is the \emph{transition kernel}
\item $O : S \to \Delta(\Omega)$ is the \emph{observation model}
\end{itemize}
The agent chooses a policy $\pi : (\Omega^*) \to A$ mapping observation histories to actions. Objective: maximize expected cumulative utility $\mathbb{E}\left[\sum_{t=0}^{T-1} \gamma^t U(a_t, s_t)\right]$ with discount factor $\gamma \in [0,1)$.
\end{definition}

\begin{notation}[Observation History]
Let $\omega_t = (o_0, o_1, \ldots, o_t)$ denote the observation history up to time $t$. The agent's policy is $\pi(\omega_{t-1}) = a_t$.
\end{notation}

\begin{definition}[Sequential Sufficiency]\label{def:sequential-sufficient}
A coordinate set $I \subseteq \{1, \ldots, n\}$ is \emph{sequentially sufficient} for $\mathcal{D}_{SEQ}$ if the optimal policy based on full observation history is the same as the optimal policy based only on coordinates in $I$ at each time step.

Formally: let $\pi^*$ be the optimal policy with full observations, and let $\pi_I^*$ be the optimal policy with observations restricted to $I$-coordinates. Then $\pi^* = \pi_I^*$ (as functions of observation history).
\end{definition}

\begin{decision}\label{dec:sequential-sufficiency-check}
\textbf{SEQUENTIAL-SUFFICIENCY-CHECK}: Given sequential decision problem $\mathcal{D}_{SEQ}$ and coordinate set $I$, is $I$ sequentially sufficient?
\end{decision}

\begin{decision}\label{dec:sequential-minsuff}
\textbf{SEQUENTIAL-MINIMUM-SUFFICIENT-SET}: Given $\mathcal{D}_{SEQ}$, find the minimal $I$ that is sequentially sufficient.
\end{decision}

\subsection{Complexity: \PSPACE-Completeness}

\begin{theorem}[SEQUENTIAL-SUFFICIENCY-CHECK is \PSPACE-complete]\label{th:sequential-sufficiency-pspace}
SEQUENTIAL-SUFFICIENCY-CHECK is \PSPACE-complete.
\end theorem}

\begin{enumerate}
\item \textbf{Membership in \PSPACE}: Optimal POMDP policy computation is in \PSPACE (Papadimitriou \& Tsitsiklis, 1987). Checking whether a coordinate set $I$ yields the same policy involves comparing value functions, which can be done in polynomial space via alternating polynomial time.

\item \textbf{\PSPACE-hardness via TQBF reduction}: Given quantified Boolean formula $\forall x_1 \exists x_2 \forall x_3 \ldots \varphi(x_1, \ldots, x_n)$, construct:
\begin{itemize}
\item $n$ time steps, one variable per step
\item At odd steps (universal quantifier): "nature" chooses the variable (adversarial transition)
\item At even steps (existential quantifier): agent chooses the variable (action)
\item Terminal utility: $\varphi(x_1, \ldots, x_n)$
\item Observation: agent sees all previous variables
\item $I$ = full observation history
\end{itemize}
Then full observation is sequentially sufficient iff the agent can guarantee $\varphi = \text{true}$ regardless of nature's choices, which holds iff the QBF is true.
\end{enumerate}

\begin{construction}[TQBF Reduction]\label{constr:tqbf-to-sequential}
Given QBF $\Phi = Q_1 x_1 Q_2 x_2 \ldots Q_n x_n\ \varphi(x_1, \ldots, x_n)$ with alternating quantifiers, construct sequential problem:
\begin{itemize}
\item Time steps $t = 1, \ldots, n$
\item State at step $t$ encodes $(x_1, \ldots, x_t)$
\item At step $t$:
  - If $Q_t = \forall$: transition is adversarial (nature chooses both next state and observation)
  - If $Q_t = \exists$: transition is controlled by agent action (choose $x_t$)
\item Terminal reward: 1 if $\varphi(x_1, \ldots, x_n) = \text{true}$, 0 otherwise
\item Full observation reveals all previous variables
\end{itemize}
Then: full observation is sequentially sufficient $\iff$ there exists a policy guaranteeing $\varphi = \text{true}$ $\iff$ $\Phi$ is true.
\end{construction}

\subsection{Tractable Subcases}

\begin{enumerate}
\item \textbf{Fully observable (MDP)}: $O$ is the identity (agent sees full state). Then the problem reduces to a stochastic (or deterministic) problem per time step. Value iteration is polynomial in $|S|, |A|, T$.

\item \textbf{Bounded horizon}: $T \leq k$. The state space grows as $O(|S|^T)$, but for fixed $T$ this is polynomial in $|S|$.

\item \textbf{Tree-structured transitions}: The transition graph has no cycles. Dynamic programming yields polynomial algorithms.

\item \textbf{Deterministic transitions}: $T(s'|a,s)$ is a point mass. The problem reduces to search over deterministic paths.
\end{enumerate}

\begin{claim}{Tractability under full observability}{prop:mdp-tractable}
If $O$ is the identity (fully observable), then SEQUENTIAL-SUFFICIENCY-CHECK reduces to checking stochastic sufficiency at each time step.
\claimstamp{P\ref{prop:mdp-tractable}}{FO}
\end{claim}

\leanmeta{\LH{CC50}}

\begin{claim}{Tractability under bounded horizon}{prop:sequential-bounded-horizon}
If $T \leq k$ (constant horizon), then SEQUENTIAL-SUFFICIENCY-CHECK is solvable in time polynomial in $|S|, |A|, k$.
\claimstamp{P\ref{prop:sequential-bounded-horizon}}{BH}
\end{claim}

Similarly, stochastic sufficiency does NOT transfer to sequential in general:

\begin{claim}{Stochastic to sequential bridge failure}{prop:stochastic-sequential-bridge-fail}
There exists a stochastic sufficient coordinate set $I$ and a sequential problem where $I$ is NOT sequentially sufficient, even when transitions are memoryless.
\claimstamp{P\ref{prop:stochastic-sequential-bridge-fail}}{CE}
\end{claim}

\leanmeta{\LH{CC56}}

The transfer conditions are:
\begin{itemize}
\item Static → Sequential: transfers iff $T = 1$ and transitions are deterministic (one-step bridge from Paper 4)
\item Stochastic → Sequential: transfers iff transitions are memoryless AND observation model is regime-compatible
\end{itemize}
